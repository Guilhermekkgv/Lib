local Linux = {}
local Flipper = require(game.ReplicatedStorage:WaitForChild("Flipper") or loadstring("https://raw.githubusercontent.com/dawid-scripts/Flipper/master/Flipper.lua")())
local TweenService = game:GetService("TweenService")

local Spring = Flipper.Spring.new
local Instant = Flipper.Instant.new

local function New(class, props)
    local obj = Instance.new(class)
    for prop, value in pairs(props or {}) do
        if prop ~= "Parent" then
            obj[prop] = value
        end
    end
    if props and props.Parent then
        obj.Parent = props.Parent
    end
    return obj
end

function Linux:CreateWindow(config)
    local ScreenGui = New("ScreenGui", {Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui"), ResetOnSpawn = false})
    
    local MainFrame = New("Frame", {
        Size = config.Size or UDim2.fromOffset(600, 400),
        Position = UDim2.new(0.5, -300, 0.5, -200),
        BackgroundColor3 = Color3.fromRGB(26, 26, 26),
        BorderSizePixel = 0,
        Active = true,
        Draggable = true,
        Parent = ScreenGui
    })
    New("UICorner", {CornerRadius = UDim.new(0, 8), Parent = MainFrame})
    
    local TopBar = New("Frame", {
        Size = UDim2.new(1, 0, 0, 40),
        BackgroundColor3 = Color3.fromRGB(32, 32, 32),
        BorderSizePixel = 0,
        Parent = MainFrame
    })
    New("UICorner", {CornerRadius = UDim.new(0, 8), Parent = TopBar})
    
    local Title = New("TextLabel", {
        Size = UDim2.new(1, -20, 1, 0),
        Position = UDim2.new(0, 10, 0, 0),
        BackgroundTransparency = 1,
        Text = config.Title or "Linux Hub",
        TextColor3 = Color3.fromRGB(240, 240, 240),
        TextSize = 16,
        FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.Bold),
        TextXAlignment = Enum.TextXAlignment.Left,
        Parent = TopBar
    })
    
    local TabHolder = New("ScrollingFrame", {
        Size = UDim2.new(0, 120, 1, -50),
        Position = UDim2.new(0, 10, 0, 50),
        BackgroundTransparency = 1,
        BorderSizePixel = 0,
        ScrollBarThickness = 0,
        CanvasSize = UDim2.new(0, 0, 0, 0),
        Parent = MainFrame
    })
    
    local ContainerCanvas = New("Frame", {
        Size = UDim2.new(1, -140, 1, -50),
        Position = UDim2.new(0, 130, 0, 50),
        BackgroundTransparency = 1,
        Parent = MainFrame
    })
    
    local ContainerHolder = New("ScrollingFrame", {
        Size = UDim2.fromScale(1, 1),
        BackgroundTransparency = 1,
        BorderSizePixel = 0,
        ScrollBarThickness = 3,
        ScrollBarImageColor3 = Color3.fromRGB(255, 255, 255),
        ScrollBarImageTransparency = 0.95,
        CanvasSize = UDim2.fromScale(0, 0),
        Parent = ContainerCanvas
    })
    New("UIPadding", {PaddingRight = UDim.new(0, 10), PaddingLeft = UDim.new(0, 1), PaddingTop = UDim.new(0, 1), PaddingBottom = UDim.new(0, 1), Parent = ContainerHolder})
    
    local Window = {Tabs = {}, Containers = {}, SelectedTab = 0, TabCount = 0}
    Window.TabHolder = TabHolder
    Window.ContainerHolder = ContainerHolder
    Window.ContainerCanvas = ContainerCanvas
    
    function Window:AddTab(config)
        self.TabCount = self.TabCount + 1
        local TabIndex = self.TabCount
        
        local Tab = {Name = config.Title}
        
        Tab.Frame = New("TextButton", {
            Size = UDim2.new(1, 0, 0, 34),
            BackgroundColor3 = Color3.fromRGB(120, 120, 120),
            BackgroundTransparency = 1,
            Text = "",
            Parent = TabHolder
        })
        New("UICorner", {CornerRadius = UDim.new(0, 6), Parent = Tab.Frame})
        
        local Icon = config.Icon and New("ImageLabel", {
            Size = UDim2.fromOffset(16, 16),
            Position = UDim2.new(0, 8, 0.5, 0),
            AnchorPoint = Vector2.new(0, 0.5),
            BackgroundTransparency = 1,
            Image = config.Icon,
            ImageColor3 = Color3.fromRGB(240, 240, 240),
            Parent = Tab.Frame
        })
        
        New("TextLabel", {
            AnchorPoint = Vector2.new(0, 0.5),
            Position = Icon and UDim2.new(0, 30, 0.5, 0) or UDim2.new(0, 12, 0.5, 0),
            Text = config.Title,
            TextColor3 = Color3.fromRGB(240, 240, 240),
            TextSize = 12,
            FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json"),
            TextXAlignment = Enum.TextXAlignment.Left,
            Size = UDim2.new(1, -12, 1, 0),
            BackgroundTransparency = 1,
            Parent = Tab.Frame
        })
        
        local ContainerLayout = New("UIListLayout", {Padding = UDim.new(0, 5), SortOrder = Enum.SortOrder.LayoutOrder})
        Tab.Container = New("ScrollingFrame", {
            Size = UDim2.fromScale(1, 1),
            BackgroundTransparency = 1,
            Parent = ContainerHolder,
            Visible = false,
            ScrollBarThickness = 3,
            ScrollBarImageColor3 = Color3.fromRGB(255, 255, 255),
            ScrollBarImageTransparency = 0.95,
            CanvasSize = UDim2.fromScale(0, 0),
        }, {ContainerLayout, New("UIPadding", {PaddingRight = UDim.new(0, 10), PaddingLeft = UDim.new(0, 1), PaddingTop = UDim.new(0, 1), PaddingBottom = UDim.new(0, 1)})})
        
        ContainerLayout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
            Tab.Container.CanvasSize = UDim2.new(0, 0, 0, ContainerLayout.AbsoluteContentSize.Y + 2)
        end)
        
        local Motor, SetTransparency = function(initial, target) 
            local motor = Flipper.SingleMotor.new(initial)
            motor:onStep(function(value) target.BackgroundTransparency = value end)
            return motor, function(value) motor:setGoal(Spring(value, {frequency = 6})) end
        end
        Tab.Motor, Tab.SetTransparency = Motor(1, Tab.Frame)
        
        Tab.Frame.MouseEnter:Connect(function() Tab.SetTransparency(Tab.Selected and 0.85 or 0.89) end)
        Tab.Frame.MouseLeave:Connect(function() Tab.SetTransparency(Tab.Selected and 0.89 or 1) end)
        Tab.Frame.MouseButton1Down:Connect(function() Tab.SetTransparency(0.92) end)
        Tab.Frame.MouseButton1Up:Connect(function() Tab.SetTransparency(Tab.Selected and 0.85 or 0.89) end)
        Tab.Frame.MouseButton1Click:Connect(function() self:SelectTab(TabIndex) end)
        
        self.Tabs[TabIndex] = Tab
        self.Containers[TabIndex] = Tab.Container
        TabHolder.CanvasSize = UDim2.new(0, 0, 0, self.TabCount * 40)
        
        if TabIndex == 1 then self:SelectTab(1) end
        
        function Tab:AddSection(title)
            local SectionFrame = New("Frame", {BackgroundTransparency = 1, Size = UDim2.new(1, 0, 0, 26), Parent = Tab.Container})
            local SectionLayout = New("UIListLayout", {Padding = UDim.new(0, 5), Parent = SectionFrame})
            local SectionContainer = New("Frame", {Size = UDim2.new(1, 0, 0, 26), Position = UDim2.fromOffset(0, 24), BackgroundTransparency = 1, Parent = SectionFrame}, {SectionLayout})
            New("TextLabel", {Text = title, TextColor3 = Color3.fromRGB(240, 240, 240), TextSize = 18, FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.SemiBold), TextXAlignment = Enum.TextXAlignment.Left, Size = UDim2.new(1, -16, 0, 18), Position = UDim2.fromOffset(0, 2), BackgroundTransparency = 1, Parent = SectionFrame})
            
            SectionLayout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
                SectionContainer.Size = UDim2.new(1, 0, 0, SectionLayout.AbsoluteContentSize.Y)
                SectionFrame.Size = UDim2.new(1, 0, 0, SectionLayout.AbsoluteContentSize.Y + 25)
            end)
            
            local Section = {Container = SectionContainer}
            
            function Section:AddToggle(config)
                local ToggleFrame = New("Frame", {Size = UDim2.new(1, -10, 0, 40), Position = UDim2.new(0, 5, 0, #SectionContainer:GetChildren() * 50), BackgroundColor3 = Color3.fromRGB(120, 120, 120), BackgroundTransparency = 0.87, Parent = SectionContainer})
                New("UICorner", {CornerRadius = UDim.new(0, 6), Parent = ToggleFrame})
                New("UIStroke", {Transparency = 0.5, Color = Color3.fromRGB(90, 90, 90), ApplyStrokeMode = Enum.ApplyStrokeMode.Border, Parent = ToggleFrame})
                
                New("TextLabel", {Text = config.Title, TextColor3 = Color3.fromRGB(240, 240, 240), TextSize = 14, FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json"), TextXAlignment = Enum.TextXAlignment.Left, Size = UDim2.new(1, -60, 1, 0), Position = UDim2.new(0, 10, 0, 0), BackgroundTransparency = 1, Parent = ToggleFrame})
                
                local ToggleSwitch = New("Frame", {Size = UDim2.new(0, 40, 0, 20), Position = UDim2.new(1, -50, 0.5, -10), BackgroundColor3 = config.Default and Color3.fromRGB(96, 205, 255) or Color3.fromRGB(120, 120, 120), Parent = ToggleFrame})
                New("UICorner", {CornerRadius = UDim.new(0, 10), Parent = ToggleSwitch})
                
                local ToggleKnob = New("Frame", {Size = UDim2.new(0, 16, 0, 16), Position = config.Default and UDim2.new(1, -18, 0.5, -8) or UDim2.new(0, 2, 0.5, -8), BackgroundColor3 = Color3.fromRGB(255, 255, 255), Parent = ToggleSwitch})
                New("UICorner", {CornerRadius = UDim.new(0, 8), Parent = ToggleKnob})
                
                local state = config.Default or false
                ToggleSwitch.InputBegan:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                        state = not state
                        TweenService:Create(ToggleKnob, TweenInfo.new(0.2, Enum.EasingStyle.Quad), {Position = state and UDim2.new(1, -18, 0.5, -8) or UDim2.new(0, 2, 0.5, -8)}):Play()
                        TweenService:Create(ToggleSwitch, TweenInfo.new(0.2, Enum.EasingStyle.Quad), {BackgroundColor3 = state and Color3.fromRGB(96, 205, 255) or Color3.fromRGB(120, 120, 120)}):Play()
                        config.Callback(state)
                    end
                end)
                
                SectionContainer.Size = UDim2.new(1, 0, 0, SectionContainer.Size.Y.Offset + 50)
            end
            
            function Section:AddDropdown(config)
                local DropdownFrame = New("Frame", {Size = UDim2.new(1, -10, 0, 40), Position = UDim2.new(0, 5, 0, #SectionContainer:GetChildren() * 50), BackgroundColor3 = Color3.fromRGB(120, 120, 120), BackgroundTransparency = 0.87, Parent = SectionContainer})
                New("UICorner", {CornerRadius = UDim.new(0, 6), Parent = DropdownFrame})
                New("UIStroke", {Transparency = 0.5, Color = Color3.fromRGB(90, 90, 90), ApplyStrokeMode = Enum.ApplyStrokeMode.Border, Parent = DropdownFrame})
                
                New("TextLabel", {Text = config.Title, TextColor3 = Color3.fromRGB(240, 240, 240), TextSize = 14, FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json"), TextXAlignment = Enum.TextXAlignment.Left, Size = UDim2.new(1, -170, 1, 0), Position = UDim2.new(0, 10, 0, 0), BackgroundTransparency = 1, Parent = DropdownFrame})
                
                local DropdownInner = New("TextButton", {Size = UDim2.fromOffset(160, 30), Position = UDim2.new(1, -10, 0.5, 0), AnchorPoint = Vector2.new(1, 0.5), BackgroundColor3 = Color3.fromRGB(160, 160, 160), BackgroundTransparency = 0.9, Text = "", Parent = DropdownFrame})
                New("UICorner", {CornerRadius = UDim.new(0, 5), Parent = DropdownInner})
                New("UIStroke", {Transparency = 0.5, Color = Color3.fromRGB(90, 90, 90), ApplyStrokeMode = Enum.ApplyStrokeMode.Border, Parent = DropdownInner})
                
                local DropdownDisplay = New("TextLabel", {Text = config.Default or config.Values[1], TextColor3 = Color3.fromRGB(240, 240, 240), TextSize = 13, FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json"), TextXAlignment = Enum.TextXAlignment.Left, Size = UDim2.new(1, -30, 0, 14), Position = UDim2.new(0, 8, 0.5, 0), AnchorPoint = Vector2.new(0, 0.5), BackgroundTransparency = 1, Parent = DropdownInner})
                New("ImageLabel", {Image = "rbxassetid://10709790948", Size = UDim2.fromOffset(16, 16), Position = UDim2.new(1, -8, 0.5, 0), AnchorPoint = Vector2.new(1, 0.5), BackgroundTransparency = 1, ImageColor3 = Color3.fromRGB(170, 170, 170), Parent = DropdownInner})
                
                local DropdownHolder = New("Frame", {Size = UDim2.fromOffset(170, 0), BackgroundColor3 = Color3.fromRGB(45, 45, 45), Parent = ScreenGui, Visible = false})
                New("UICorner", {CornerRadius = UDim.new(0, 7), Parent = DropdownHolder})
                New("UIStroke", {Color = Color3.fromRGB(35, 35, 35), ApplyStrokeMode = Enum.ApplyStrokeMode.Border, Parent = DropdownHolder})
                
                local DropdownScroll = New("ScrollingFrame", {Size = UDim2.new(1, -5, 1, -10), Position = UDim2.fromOffset(5, 5), BackgroundTransparency = 1, ScrollBarThickness = 4, ScrollBarImageColor3 = Color3.fromRGB(255, 255, 255), ScrollBarImageTransparency = 0.95, CanvasSize = UDim2.fromScale(0, 0), Parent = DropdownHolder})
                local DropdownListLayout = New("UIListLayout", {Padding = UDim.new(0, 3), Parent = DropdownScroll})
                
                local Dropdown = {Values = config.Values, Value = config.Default or config.Values[1], Opened = false, Buttons = {}}
                
                local function UpdateHolderPosition()
                    local posY = DropdownInner.AbsolutePosition.Y + 35
                    if game.Workspace.CurrentCamera.ViewportSize.Y - posY < 392 then
                        posY = DropdownInner.AbsolutePosition.Y - 392
                    end
                    DropdownHolder.Position = UDim2.fromOffset(DropdownInner.AbsolutePosition.X - 1, posY)
                end
                
                DropdownInner:GetPropertyChangedSignal("AbsolutePosition"):Connect(UpdateHolderPosition)
                
                local function BuildDropdownList()
                    for _, child in pairs(DropdownScroll:GetChildren()) do
                        if not child:IsA("UIListLayout") then child:Destroy() end
                    end
                    
                    for _, value in pairs(Dropdown.Values) do
                        local Button = New("TextButton", {Size = UDim2.new(1, -5, 0, 32), BackgroundTransparency = 1, Text = "", Parent = DropdownScroll})
                        local Selector = New("Frame", {Size = UDim2.fromOffset(4, 14), BackgroundColor3 = Color3.fromRGB(96, 205, 255), Position = UDim2.fromOffset(-1, 16), AnchorPoint = Vector2.new(0, 0.5), Parent = Button})
                        New("UICorner", {CornerRadius = UDim.new(0, 2), Parent = Selector})
                        local Label = New("TextLabel", {Text = value, TextColor3 = Color3.fromRGB(200, 200, 200), TextSize = 13, FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json"), TextXAlignment = Enum.TextXAlignment.Left, Size = UDim2.fromScale(1, 1), Position = UDim2.fromOffset(10, 0), BackgroundTransparency = 1, Parent = Button})
                        
                        local Selected = Dropdown.Value == value
                        local BackMotor, SetBackTransparency = Motor(Selected and 0.89 or 1, Button)
                        local SelMotor, SetSelTransparency = Motor(Selected and 0 or 1, Selector)
                        local SizeMotor = Flipper.SingleMotor.new(Selected and 14 or 6)
                        SizeMotor:onStep(function(val) Selector.Size = UDim2.new(0, 4, 0, val) end)
                        
                        Button.MouseEnter:Connect(function() SetBackTransparency(Selected and 0.85 or 0.89) end)
                        Button.MouseLeave:Connect(function() SetBackTransparency(Selected and 0.89 or 1) end)
                        Button.MouseButton1Down:Connect(function() SetBackTransparency(0.92) end)
                        Button.MouseButton1Up:Connect(function() SetBackTransparency(Selected and 0.85 or 0.89) end)
                        Button.MouseButton1Click:Connect(function()
                            Dropdown.Value = value
                            DropdownDisplay.Text = value
                            for _, btn in pairs(Dropdown.Buttons) do btn:Update() end
                            Dropdown:Close()
                            config.Callback(value)
                        end)
                        
                        local Btn = {
                            Update = function()
                                Selected = Dropdown.Value == value
                                SetBackTransparency(Selected and 0.89 or 1)
                                SetSelTransparency(Selected and 0 or 1)
                                SizeMotor:setGoal(Spring(Selected and 14 or 6, {frequency = 6}))
                            end
                        }
                        Dropdown.Buttons[Button] = Btn
                    end
                    
                    DropdownScroll.CanvasSize = UDim2.fromOffset(0, DropdownListLayout.AbsoluteContentSize.Y)
                    DropdownHolder.Size = UDim2.fromOffset(170, math.min(DropdownListLayout.AbsoluteContentSize.Y + 10, 392))
                    UpdateHolderPosition()
                end
                
                Dropdown.Open = function()
                    Dropdown.Opened = true
                    DropdownHolder.Visible = true
                    TweenService:Create(DropdownHolder, TweenInfo.new(0.2, Enum.EasingStyle.Quart), {Size = UDim2.fromOffset(170, math.min(DropdownListLayout.AbsoluteContentSize.Y + 10, 392))}):Play()
                end
                
                Dropdown.Close = function()
                    Dropdown.Opened = false
                    TweenService:Create(DropdownHolder, TweenInfo.new(0.2, Enum.EasingStyle.Quart), {Size = UDim2.fromOffset(170, 0)}):Play()
                    task.wait(0.2)
                    DropdownHolder.Visible = false
                end
                
                DropdownInner.MouseButton1Click:Connect(function()
                    if Dropdown.Opened then Dropdown:Close() else Dropdown:Open() end
                end)
                
                game:GetService("UserInputService").InputBegan:Connect(function(input)
                    if (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) and Dropdown.Opened then
                        local mouse = game.Players.LocalPlayer:GetMouse()
                        local absPos, absSize = DropdownHolder.AbsolutePosition, DropdownHolder.AbsoluteSize
                        if mouse.X < absPos.X or mouse.X > absPos.X + absSize.X or mouse.Y < absPos.Y or mouse.Y > absPos.Y + absSize.Y then
                            Dropdown:Close()
                        end
                    end
                end)
                
                BuildDropdownList()
                SectionContainer.Size = UDim2.new(1, 0, 0, SectionContainer.Size.Y.Offset + 50)
            end
            
            return Section
        end
        
        return Tab
    end
    
    function Window:SelectTab(tabIndex)
        local tab = self.Tabs[tabIndex]
        if not tab then return end
        
        self.SelectedTab = tabIndex
        for _, t in pairs(self.Tabs) do
            t.SetTransparency(1)
            t.Selected = false
        end
        tab.SetTransparency(0.89)
        tab.Selected = true
        
        for _, c in pairs(self.Containers) do c.Visible = false end
        self.Containers[tabIndex].Visible = true
    end
    
    return Window
end

return Linux
